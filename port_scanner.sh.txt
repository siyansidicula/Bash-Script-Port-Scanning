#!/bin/bash


if [ $# -ne 1 ]; then
    echo "Usage: $0 <host>"
    exit 1
fi

host=$1

echo "Scanning common ports on $host..."

for port in 22 25 53 80 110 143 443 3306 8080; do
    timeout 1 bash -c "echo >/dev/tcp/$host/$port" 2>/dev/null

    if [ $? -eq 0 ]; then
        echo "Port $port is OPEN"
    else
        echo "Port $port is CLOSED"
    fi
done
